pipeline {
    agent any

    stages {
	def varmaven1 = tool name: 'maven', type: 'maven'
        stage ('Compile Stage') {
		tool name: 'maven', type: 'maven'

            steps {
					
                    sh "${varmaven1}/bin/mvn package"
                
            }
        }

        stage ('Testing Stage') {

            steps {
               
                    sh "${varmaven1}/bin/mvn test"
                }
            }
        
        stage ('Deployment Stage') {
            steps {
                
                    sh "${varmaven1}/bin/mvn deploy"
            }
        }
    }
}
