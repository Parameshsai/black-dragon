pipeline {
    agent any

    stages {
        stage ('Compile Stage') {
		tool name: 'maven', type: 'maven'

            steps {
					def varmaven1 = tool name: 'maven', type: 'maven'
                    sh "${varmaven1}/bin/mvn clean compile"
                
            }
        }

        stage ('Testing Stage') {

            steps {
                def varmaven2 = tool name: 'maven', type: 'maven'
                    sh "${varmaven2}/bin/mvn test"
                }
            }
        
        stage ('Deployment Stage') {
            steps {
                def varmaven3 = tool name: 'maven', type: 'maven'
                    sh "${varmaven3}/bin/mvn deploy"
            }
        }
    }
}
